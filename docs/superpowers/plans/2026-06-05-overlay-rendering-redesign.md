# Overlay Rendering Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stabilize the project and replace the monolithic video overlay canvas with a tested, modular, transparent HUD rendering subsystem.

**Architecture:** Keep Canvas as the rendering surface, but split frame telemetry, mini-map geometry, widget rendering, and screenshot capture into focused modules under `src/components/VideoOverlay/overlay/`. `OverlayCanvas.tsx` becomes the host for sizing, refs, animation, and renderer orchestration.

**Tech Stack:** React 19, TypeScript 6, Vite 8, Electron 42, Zustand, Ant Design, Canvas 2D, Vitest with jsdom.

Create `src/components/VideoOverlay/overlay/overlayTelemetry.ts`:

```ts
import type { LapData, TrackPoint } from '../../../types';
import { calculateAcceleration, getPointAtTime } from '../../../utils/trackCalc';
import type { AccelerationSample, OverlayTelemetry } from './types';

interface DeriveOverlayTelemetryInput {
  points: TrackPoint[];
  laps: LapData[];
  videoStartTime: Date | null;
  currentTime: number;
  timeOffset: number;
  accelerations?: AccelerationSample[];
}

function getLapTiming(
  currentPoint: TrackPoint,
  laps: LapData[],
): { lapTime: number | null; lapNumber: number | null } {
  const pointTime = currentPoint.timestamp.getTime();

  for (const lap of laps) {
    const startTime = lap.startTime.getTime();
    const endTime = lap.endTime.getTime();
    if (pointTime >= startTime && pointTime <= endTime) {
      return {
        lapTime: (pointTime - startTime) / 1000,
        lapNumber: lap.lapNumber,
      };
    }
  }

  const lastLap = laps[laps.length - 1];
  if (lastLap && pointTime > lastLap.endTime.getTime()) {
    return {
      lapTime: (pointTime - lastLap.endTime.getTime()) / 1000,
      lapNumber: lastLap.lapNumber + 1,
    };
  }

  return { lapTime: null, lapNumber: null };
}

export function deriveOverlayTelemetry(input: DeriveOverlayTelemetryInput): OverlayTelemetry | null {
  const { points, laps, videoStartTime, currentTime, timeOffset } = input;
  if (points.length === 0 || !videoStartTime) return null;

  const videoTime = new Date(videoStartTime.getTime() + currentTime * 1000);
  const match = getPointAtTime(points, videoTime, timeOffset);
  if (!match) return null;

  const accelerations = input.accelerations ?? calculateAcceleration(points);
  const acceleration = accelerations[match.index] ?? { longitudinal: 0, lateral: 0 };
  const lapTiming = getLapTiming(match.point, laps);

  return {
    videoTime,
    currentPoint: match.point,
    currentIndex: match.index,
    acceleration,
    ...lapTiming,
  };
}
```

```ts
import type { TrackPoint } from '../../../types';
import type { OverlayPoint, OverlayRect } from './types';

export interface TrackProjection {
  project: (point: TrackPoint) => OverlayPoint;
  bounds: {
    minLat: number;
    maxLat: number;
    minLon: number;
    maxLon: number;
  };
}

export function createTrackProjection(
  points: TrackPoint[],
  rect: OverlayRect,
  padding = 8,
): TrackProjection {
  const lats = points.map((p) => p.lat);
  const lons = points.map((p) => p.lon);
  const minLat = Math.min(...lats);
  const maxLat = Math.max(...lats);
  const minLon = Math.min(...lons);
  const maxLon = Math.max(...lons);
  const latRange = maxLat - minLat || 0.001;
  const lonRange = maxLon - minLon || 0.001;
  const drawWidth = Math.max(rect.width - padding * 2, 1);
  const drawHeight = Math.max(rect.height - padding * 2, 1);
  const scale = Math.min(drawWidth / lonRange, drawHeight / latRange);
  const offsetX = (drawWidth - lonRange * scale) / 2;
  const offsetY = (drawHeight - latRange * scale) / 2;

  return {
    bounds: { minLat, maxLat, minLon, maxLon },
    project(point) {
      return {
        x: rect.x + padding + offsetX + (point.lon - minLon) * scale,
        y: rect.y + padding + drawHeight - offsetY - (point.lat - minLat) * scale,
      };
    },
  };
}

export function splitTrackProgress(
  points: TrackPoint[],
  currentIndex: number,
): { completed: TrackPoint[]; remaining: TrackPoint[] } {
  if (points.length === 0) return { completed: [], remaining: [] };
  const clamped = Math.max(0, Math.min(currentIndex, points.length - 1));
  return {
    completed: points.slice(0, clamped + 1),
    remaining: points.slice(clamped),
  };
}

function bearing(lat1: number, lon1: number, lat2: number, lon2: number): number {
  const dLon = ((lon2 - lon1) * Math.PI) / 180;
  const la1 = (lat1 * Math.PI) / 180;
  const la2 = (lat2 * Math.PI) / 180;
  const y = Math.sin(dLon) * Math.cos(la2);
  const x = Math.cos(la1) * Math.sin(la2) - Math.sin(la1) * Math.cos(la2) * Math.cos(dLon);
  return ((Math.atan2(y, x) * 180) / Math.PI + 360) % 360;
}

export function getHeadingForPoint(points: TrackPoint[], index: number): number {
  const current = points[index];
  if (!current) return 0;
  if (typeof current.heading === 'number') return current.heading;

  const next = points[Math.min(index + 1, points.length - 1)];
  const prev = points[Math.max(index - 1, 0)];
  if (next && next !== current) return bearing(current.lat, current.lon, next.lat, next.lon);
  if (prev && prev !== current) return bearing(prev.lat, prev.lon, current.lat, current.lon);
  return 0;
}

export function buildDirectionArrow(center: OverlayPoint, heading: number, size: number): OverlayPoint[] {
  const angle = ((heading - 90) * Math.PI) / 180;
  const tip = {
    x: center.x + Math.cos(angle) * size,
    y: center.y + Math.sin(angle) * size,
  };
  const leftAngle = angle + Math.PI * 0.78;
  const rightAngle = angle - Math.PI * 0.78;

  return [
    tip,
    {
      x: center.x + Math.cos(leftAngle) * size * 0.65,
      y: center.y + Math.sin(leftAngle) * size * 0.65,
    },
    {
      x: center.x + Math.cos(rightAngle) * size * 0.65,
      y: center.y + Math.sin(rightAngle) * size * 0.65,
    },
  ];
}
```

```tsx
    onCanvasReady?.(canvasRef.current);
    return () => onCanvasReady?.(null);

    inputRef.current = {
      width: canvasSizeRef.current.width,
      height: canvasSizeRef.current.height,
      points: gpxPoints,
      laps,
      videoStartTime,
      currentTime,
      timeOffset,
      overlayConfig,
      accelerations: gpxPoints.length > 2 ? calculateAcceleration(gpxPoints) : [],
      maxSpeed: Math.max(...gpxPoints.map((p) => p.speed), 1),
    };
  }, [currentTime, gpxPoints, laps, overlayConfig, timeOffset, videoStartTime]);

  useEffect(() => {
    const container = containerRef.current;
    const canvas = canvasRef.current;
    if (!container || !canvas) return;

    const observer = new ResizeObserver((entries) => {
      const entry = entries[0];
      if (!entry) return;
      const { width, height } = entry.contentRect;
      const dpr = window.devicePixelRatio || 1;
      canvas.width = Math.max(1, Math.round(width * dpr));
      canvas.height = Math.max(1, Math.round(height * dpr));
      canvas.style.width = `${width}px`;
      canvas.style.height = `${height}px`;
      canvasSizeRef.current = { width, height };
      if (inputRef.current) {
        inputRef.current = { ...inputRef.current, width, height };
      }
    });

    observer.observe(container);
    return () => observer.disconnect();
  }, []);

  useEffect(() => {
    function drawFrame() {
      const canvas = canvasRef.current;
      const input = inputRef.current;
      if (!canvas || !input) {
        animRef.current = requestAnimationFrame(drawFrame);
        return;
      }

      const ctx = canvas.getContext('2d');
      if (!ctx) {
        animRef.current = requestAnimationFrame(drawFrame);
        return;
      }

      const dpr = window.devicePixelRatio || 1;
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      const telemetry = deriveOverlayTelemetry(input);
      renderOverlayFrame(ctx, input, telemetry);
      animRef.current = requestAnimationFrame(drawFrame);
    }

    animRef.current = requestAnimationFrame(drawFrame);
    return () => cancelAnimationFrame(animRef.current);
  }, []);

  return (

      ref={containerRef}
      style={{
        position: 'absolute',
        top: 0,
        left: 0,
        width: '100%',
        height: '100%',
        pointerEvents: 'none',
      }}
    >
      <canvas ref={canvasRef} />
    </div>
  );
```

```ts
    throw new Error('视频帧尚未准备好');
  }

  const output = document.createElement('canvas');
  output.width = video.videoWidth;
  output.height = video.videoHeight;
  const ctx = output.getContext('2d');
  if (!ctx) {
    throw new Error('无法创建截图画布');
  }

  ctx.drawImage(video, 0, 0, output.width, output.height);
  ctx.drawImage(overlayCanvas, 0, 0, output.width, output.height);

  const link = document.createElement('a');
  link.href = output.toDataURL('image/png');
  link.download = filename;
  link.click();
}
```
